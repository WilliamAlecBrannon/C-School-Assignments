#include "stdafx.h"
#include <iostream>

using namespace std;

class FuelGauge
{
private:
	int currentFuel;
	int maxFuel;
public:
	FuelGauge(const int = 0, const int = 15);
	FuelGauge(const FuelGauge&);
	int reportFuelLevel();
	int reportMaxFuel();
	void incrementFuel();
	void decrementFuel();
};

FuelGauge::FuelGauge(const int cFuel,const int mFuel) {
	currentFuel = cFuel;
	maxFuel = mFuel;
}

FuelGauge::FuelGauge(const FuelGauge& fGauge) {
	currentFuel = fGauge.currentFuel;
	maxFuel = fGauge.maxFuel;
}

int FuelGauge::reportFuelLevel() {
	return currentFuel;
}

int FuelGauge::reportMaxFuel() {
	return maxFuel;
}

void FuelGauge::incrementFuel() {
	if (currentFuel < maxFuel) {
		currentFuel++;
	}
}

void FuelGauge::decrementFuel() {
	if (currentFuel > 0) {
		currentFuel--;
	}
}

class Odometer {
private:
	FuelGauge fGauge;
	int mileage;
	int fuelEconomy;
public:
	Odometer(const int cFuel, const int mFuel, const int mile, const int fEconomy);
	Odometer(const Odometer&);
	int reportMileage();
	int reportFuel();
	void fillFuel();
	void driveMile();
};

Odometer::Odometer(const int cFuel, const int mFuel, const int mile, const int fEconomy) {
	fGauge = FuelGauge(cFuel, mFuel);
	mileage = mile;
	fuelEconomy = fEconomy;
}

Odometer::Odometer(const Odometer& oMeter){
	fGauge = oMeter.fGauge;
	mileage = oMeter.mileage;
	fuelEconomy = oMeter.fuelEconomy;
}

int Odometer::reportMileage() {
	return mileage;
}

int Odometer::reportFuel() {
	return fGauge.reportFuelLevel();
}

void Odometer::fillFuel() {
	int max = fGauge.reportMaxFuel();
	int current = fGauge.reportFuelLevel();
	cout << "Current Mileage: " << mileage << endl;
	cout << "Current Fuel: " << fGauge.reportFuelLevel() << endl;
	for (int i = current; i < max; i++) {
		fGauge.incrementFuel();
		cout << "Current Mileage: " << mileage << endl;
		cout << "Current Fuel: " << fGauge.reportFuelLevel() << endl;
	}
}

void Odometer::driveMile() {
	mileage++;
	if (mileage > 999999) {
		mileage = 0;
	}
	if (mileage % fuelEconomy == 0) {
		fGauge.decrementFuel();
	}
	cout << "Current Mileage: " << mileage << endl;
	cout << "Current Fuel: " << fGauge.reportFuelLevel() << endl;
}

int main() {
	Odometer tester = Odometer(0, 15, 0, 24);
	tester.fillFuel();
	int current = tester.reportFuel();
	while (current > 0) {
		tester.driveMile();
		current = tester.reportFuel();
	}
	system("pause");
	return 0;
}
